requirejs工程化,gulp自动化,browsersync实时刷新

  • browsersync实时刷新
  • gulp自动化
  • requirejs工程化

npm install requirejs

在www/src/js/lib下
01 npm install -g bower
02 bower install jquery
03 bower install requirejs
这样就出现bower_components文件夹下面有jquery和requirejs

配置main.js文件,这是app下index.js的配置文件，即入口文件
requirejs.config({
base: “./src/js”,
paths: {
‘jquery’: ‘lib/bower_components/jquery/dist/jquery.min’
});

requirejs([‘app/index’]);

配置build.js文件
({
baseUrl: “.”,
paths: {
‘jquery’: ‘lib/bower_components/jquery/dist/jquery.min’
},
name: “main”,
out: “../../dist/js/index.merge.min.js”
})

//执行r.js -o build.js 合并功能
r.js的build.js可以把所有的JS合并成dist下index.merge.min.js
后面这个功能与gulp uglify一起使用来合并压缩所有的JS为一个文件

在www下npm install -g gulp
出现node_modules下gulp
继续npm install gulp-requirejs
npm install gulp-minify-css
npm install gulp-uglify
npm install gulp-concat
npm install gulp-rename
npm install gulp-clean
上面可以用cnpm http://www.ydcss.com/archives/18#lesson5

新建gulpfile.js
var gulp = require(‘gulp’);
var rjs = require(‘gulp-requirejs’);

var minifycss = require(‘gulp-minify-css’),
uglify = require(‘gulp-uglify’),
concat = require(‘gulp-concat’),
rename = require(‘gulp-rename’),
clean = require(‘gulp-clean’);

gulp.task(‘css-index’, function() {
gulp.src(“./src/css/*.css”)
.pipe(concat(‘index.merge.min.css’))
.pipe(minifycss())
.pipe(gulp.dest(‘./dist/css/‘));
});

gulp.task(‘js-index’, function() {
rjs({
baseUrl: “./src/js”,
paths: {
‘jquery’: ‘lib/bower_components/jquery/dist/jquery.min’
},
name: “main”,
out: “index.merge.min.js”
})
.pipe(uglify())
.pipe(gulp.dest(“./dist/js”));

gulp.src("./src/js/lib/bower_components/requirejs/require.js")
    .pipe(uglify())
    .pipe(rename({
        suffix: '.min'
    }))
    .pipe(gulp.dest('./dist/js'));

});

var browserSync = require(‘browser-sync’).create();

gulp.task(‘watch’, [‘js-index’, ‘css-index’], function() {
browserSync.reload();
});

gulp.task(‘server’, function() {
browserSync.init({
server: {
baseDir: “./“
}
});

gulp.watch(['./src/**/*.css', './src/**/*.js', './*.html'], ['watch']);

});

//CSS和JS压缩合并这两个任务组合起来 执行gulp
gulp.task(‘default’, [‘css-index’, ‘js-index’]);

执行npm install -g browser-sync
browser-sync -h
browser-sync start –server –file=”*”
当更改源码时，游览器自动更新，但以上命令符只对HTML生效
要想把这个继承到CSS和JS改动上，需要在gulpfile.js里配置

官方文档www.browsersync.io/docs/gulp
执行gulp server会自动启动3000端口的页面
